%% ALL USED FUNCTIONS

% Function to estimate speed sensor parameters
function [mean_y, variance_y, C] = speedSensorEstimation(data, v)
    mean_y = mean(data, 2);
    variance_y = (data - mean_y) * (data - mean_y)' / (length(data) - 1);
    C = mean_y / v;
end

% Kalman Filter Function
function [X, P, V, P_pred_v] = kalmanFilter(Y, x_0, P_0, A, Q, H, R)
    % Parameters
    N = size(Y, 2); % Number of time steps
    n = size(x_0, 1); % Dimensionality of the state vector
    m = size(H, 1); % Dimensionality of the measurement vector
    
    % Data allocation
    X = zeros(n, N); % Estimated state vector sequence
    P = zeros(n, n, N); % Filter error covariance sequence
    V = zeros(m, N);
    P_pred_v = zeros(n, n, N);

    % Initialize state estimate and covariance
    X(:, 1) = x_0; % Prior mean
    P(:, :, 1) = P_0; % Prior covariance
    
    % Kalman filtering loop
    for t = 2:N+1
        % Prediction step
        x = X(:, t-1);
        [x_pred, P_pred] = linearPrediction(x, P(:, :, t-1), A, Q);
        P_pred_v(:,:,t) = P_pred;
        
        % Update step
        y = Y(:, t-1); % Measurement at time t
        [x, P(:, :, t), v_step] = linearUpdate(x_pred, P_pred, y, H, R);
        
        % Save estimates
        X(:, t) = x;
        V(:, t) = v_step;
    end

    % Remove initial states and covariance
    X = X(:, 2:end);
    P = P(:, :, 2:end);
end

% Linear Prediction Function
function [x, P] = linearPrediction(x, P, A, Q)
    % Predicted state mean
    x = A * x;
    % Predicted state covariance
    P = A * P * A' + Q;
end

% Linear Update Function
function [x, P, residual] = linearUpdate(x, P, y, H, R)
    % Measurement residual
    residual = y - H * x;
    % Kalman gain
    K = P * H' * inv(H * P * H' + R);
    % Updated state mean
    x = x + K * residual;
    % Updated state covariance
    P = (eye(size(P)) - K * H) * P;
end

% Function to generate linear state sequence
function X = genLinearStateSequence(x_0, P_0, A, Q, N)
    % Initialize state vector sequence
    n = size(x_0, 1);
    X = zeros(n, N+1);
    % Generate initial state from the prior
    X(:, 1) = mvnrnd(x_0, P_0)';
    % Generate subsequent states using the process model
    for t = 2:N+1
        X(:, t) = A * X(:, t-1) + mvnrnd(zeros(n, 1), Q)';
    end
end

% Function to generate linear measurement sequence
function Y = genLinearMeasurementSequence(X, H, R)
    m = size(H, 1); % Dimensionality of the observation
    N = size(X, 2) - 1; % Number of observations
    Y = zeros(m, N);
    for k = 1:N
        noise = mvnrnd(zeros(m, 1), R)';
        Y(:, k) = H * X(:, k+1) + noise;
    end
end
