%% 2 Non-linear Kalman filtering

%a
N = 100; % time steps
T = 1; % Sampling time

%Intial prior
x_0 = [0;     0;     20;     0;      5*pi/180];
P_0 = diag([10^2    10^2     2^2     (pi/180)^2      (pi/180)^2]);

%PROCESS NOISE PARAMETERS (always the same for all cases)
sigma_W = pi/180;
sigma_V = 1;
% Process disturbances
Tau=[0 0 1 0 0;0 0 0 0 1]';
Q=Tau*[sigma_V^2 0;0 sigma_W^2]*Tau'; % from 3.1.1 


%sensors positions
sensor_1 = [-200; 100];
sensor_2 = [-200; -100];


%Motion model function f and process function h
f = @(x) coordinatedTurnMotion(x, T);
h = @(x) dualBearingMeasurement(x, sensor_1, sensor_2);


%CASE 1 for measurements noise
sigma_phi_1 = (2*pi)/180;
sigma_phi_2 = (2*pi)/180;
% Measurement noise
R_1 = [sigma_phi_1^2 0; 0 sigma_phi_2^2];

%generate measurements and true pos (states)
true_pos = genNonLinearStateSequence(x_0, P_0, f, Q, N);
Measurements_pos = genNonLinearMeasurementSequence(true_pos, h, R_1);

%transform measurments to positions with the help of getPosFromMeasurement
p=[];
for case_n=1:N
    [measurement_x, measurement_y] = getPosFromMeasurement(Measurements_pos(1,case_n), Measurements_pos(2,case_n), sensor_1, sensor_2);
    p = [p;[measurement_x, measurement_y]];
end


%ESTIMATE FOR CASE 1
[x_est_EKF, P_est_EKF] = nonLinearKalmanFilter(Measurements_pos, x_0, P_0, f, Q, h, R_1, 'EKF');
[x_est_UKF, P_est_UKF] = nonLinearKalmanFilter(Measurements_pos, x_0, P_0, f, Q, h, R_1, 'UKF');
[x_est_CKF, P_est_CKF] = nonLinearKalmanFilter(Measurements_pos, x_0, P_0, f, Q, h, R_1, 'CKF');

plot(x_est_EKF(1,:), x_est_EKF(2,:),'LineWidth',2);
plot(true_pos(1,:), true_pos(2,:), 'LineWidth',2); %first state (row) is x, second row is y
plot(p(:,1), p(:,2), '--', 'Color', 'k', 'LineWidth',2); %first column is x, second column is y 

for case_n = 1:5:length(x_est_EKF)
    elipse_3sigma = sigmaEllipse2D(x_est_EKF(1:2,case_n), P_est_EKF(1:2,1:2,case_n), lvl, N);

R_s = {R_1, R_2, R_3};

for case_n = 1:3     
    for type = 1:3 
        est_error    = [];
        for imc = 1:MC_n
            % Simulate state sequence
            X = genNonLinearStateSequence(x_0, P_0, f, Q, N);
            % Simulate measurements
            Y = genNonLinearMeasurementSequence(X, h, R_s{case_n});
            % Run Kalman filter (you need to run all three, for comparison)
            [xf,Pf,xp,Pp] = nonLinearKalmanFilter(Y,x_0,P_0,f,Q,h,R_s{case_n}, types{type});
            % Save the estimation errors and the prediction errors!
            error = [X(1,2:end); X(2,2:end)] - [xf(1,:); xf(2,:)];
            est_error = [est_error ; error(1:2, :)];
        end

        var_error_x = var(est_error(:, 1));
        mean_error_x = mean(est_error(:, 1));


        var_error_y = var(est_error(:, 1));
        mean_error_y = mean(est_error(:, 1));

        figure;
        histogram(est_error(:, 1), 'Normalization','pdf');
        hold on;

        data_points_x = -5*sqrt(var_error_x):0.1:5*sqrt(var_error_x);
        distribution_x = normpdf(data_points_x, mean_error_x , sqrt(var_error_x));
        plot(data_points_x,distribution_x);
        %xlim([-5*sqrt(var_error_x), 5*sqrt(var_error_x)])
        legend('Error estimation x', 'Gaussian distribution estimation of the errors x');
        title(['x Error, R ='  num2str(case_n) ', Type =' num2str(type)])

        figure;
        histogram(est_error(2, :), 'Normalization','pdf');
        hold on;
        data_points_y = -5*sqrt(var_error_y):0.1:5*sqrt(var_error_y);
        distribution_y = normpdf(data_points_x, mean_error_y, sqrt(var_error_y));
        plot(data_points_y,distribution_y);
        legend('Error estimation y', 'Gaussian distribution estimation of the errors y');
        title(['y Error, R ='  num2str(case_n) ', Type ='  num2str(type)])
    end

end

function [fx, Fx] = coordinatedTurnMotion(x, T)
%COORDINATEDTURNMOTION calculates the predicted state using a coordinated
%turn motion model, and also calculated the motion model Jacobian
%
%Input:
%   x           [5 x 1] state vector
%   T           [1 x 1] Sampling time
%
%Output:
%   fx          [5 x 1] motion model evaluated at state x
%   Fx          [5 x 5] motion model Jacobian evaluated at state x
%
% NOTE: the motion model assumes that the state vector x consist of the
% following states:
%   px          X-position
%   py          Y-position
%   v           velocity
%   phi         heading
%   omega       turn-rate

% Your code for the motion model here
% fx = ...

% Unpack state vector
xp = x(1);
yp = x(2);
v = x(3);
theta = x(4);
omega = x(5);

% Compute motion model vector
fx = [
    xp + T*v*cos(theta);
    yp + T*v*sin(theta);
    v;
    theta + T*omega;
    omega
];

%Check if the Jacobian is requested by the calling function
if nargout > 1
    % Your code for the motion model Jacobian here
    % Fx = ...
    Fx = [1   0   T*cos(theta)   -T*v*sin(theta)   0;
          0   1   T*sin(theta)    T*v*cos(theta)   0;
          0   0       1               0            0;
          0   0       0               1            T;
          0   0       0               0            1;]
end

end



function [hx, Hx] = dualBearingMeasurement(x, s1, s2)
%DUOBEARINGMEASUREMENT calculates the bearings from two sensors, located in 
%s1 and s2, to the position given by the state vector x. Also returns the
%Jacobian of the model at x.
%
%Input:
%   x           [n x 1] State vector, the two first element are 2D position
%   s1          [2 x 1] Sensor position (2D) for sensor 1
%   s2          [2 x 1] Sensor position (2D) for sensor 2
%
%Output:
%   hx          [2 x 1] measurement vector
%   Hx          [2 x n] measurement model Jacobian
%
% NOTE: the measurement model assumes that in the state vector x, the first
% two states are X-position and Y-position.

% Your code here


n = size(x,1);

x_s1 = s1(1);
x_s2 = s2(1);
y_s1 = s1(2);
y_s2 = s2(2);


hx= [atan2(x(2,:)- y_s1, x(1,:)- x_s1);
      atan2(x(2,:)- y_s2, x(1,:)- x_s2)];
    
Hx=[-(x(2,:)- y_s1) / ((x(1,:)- x_s1)^2 + (x(2,:)- y_s1)^2),           (x(1,:)- x_s1) / ((x(1,:)- x_s1)^2 + (x(2,:)- y_s1)^2),       zeros(1,n-2);
    -(x(2,:)- y_s2) / ((x(1,:)- x_s2)^2 + (x(2,:)- y_s2)^2),            (x(1,:)- x_s2) / ((x(1,:)- x_s2)^2 + (x(2,:)- y_s2)^2),       zeros(1,n-2)];


end

function X = genNonLinearStateSequence(x_0, P_0, f, Q, N)
%GENNONLINEARSTATESEQUENCE generates an N+1-long sequence of states using a 
%    Gaussian prior and a nonlinear Gaussian process model
%
%Input:
%   x_0         [n x 1] Prior mean
%   P_0         [n x n] Prior covariance
%   f           Motion model function handle
%               [fx,Fx]=f(x) 
%               Takes as input x (state), 
%               Returns fx and Fx, motion model and Jacobian evaluated at x
%               All other model parameters, such as sample time T,
%               must be included in the function
%   Q           [n x n] Process noise covariance
%   N           [1 x 1] Number of states to generate
%
%Output:
%   X           [n x N+1] State vector sequence
%

% Your code here
n = size(x_0, 1);

X = zeros(size(x_0, 1), N+1);

% Generate initial state from the prior
X(:, 1) = mvnrnd(x_0, P_0)';

for k = 2:N+1
    % Generate process noise sample
    w_k = mvnrnd(zeros(n,1), Q)';
    
    % Evaluate motion model and Jacobian at current state
    [fx, Fx] = f(X(:,k-1));
    
    % Add process noise to the propagated state
    X(:,k) = fx + w_k;
end


end


function Y = genNonLinearMeasurementSequence(X, h, R)
%GENNONLINEARMEASUREMENTSEQUENCE generates ovservations of the states 
% sequence X using a non-linear measurement model.
%
%Input:
%   X           [n x N+1] State vector sequence
%   h           Measurement model function handle
%   h           Measurement model function handle
%               [hx,Hx]=h(x) 
%               Takes as input x (state) 
%               Returns hx and Hx, measurement model and Jacobian evaluated at x
%   R           [m x m] Measurement noise covariance
%
%Output:
%   Y           [m x N] Measurement sequence
%

% Your code here

N = size(X, 2) - 1; % Number of observations

% Extract number of measurements
m = size(R, 1);

% Initialize measurement sequence 
Y = zeros(m, N);

% Generate measurements for each state
for k = 1:N
    % Generate measurement noise sample
    noise = mvnrnd(zeros(m, 1), R)';
    
    % Evaluate measurement model and Jacobian at current state
    [hx, Hx] = h(X(:,k+1));
        
    % Compute measurement using measurement model
    Y(:,k) = hx + noise;
end

end


function [SP,W] = sigmaPoints(x, P, type)
% SIGMAPOINTS computes sigma points, either using unscented transform or
% using cubature.
%
%Input:
%   x           [n x 1] Prior mean
%   P           [n x n] Prior covariance
%
%Output:
%   SP          [n x 2n+1] UKF, [n x 2n] CKF. Matrix with sigma points
%   W           [1 x 2n+1] UKF, [1 x 2n] UKF. Vector with sigma point weights 


    % Determine n
    n = length(x);
    switch type        
        case 'UKF'        
            % SIGMA Unscented Transform
             SP = zeros(n,2*n+1);
             W = zeros(1,2*n+1);
             
             % Sigma point  x^(0) & weight w0
             SP(:,1) = x; %sigma 0  x^(0)
             W(1) = 1 - (n/3); %the only design parameter w0. x is assumed to be gaussian and therefore w0 is given this value from lecture notes
             % Get the other sigma points
             P_sqrt = chol(P,'lower');
             for i = 2:(n+1)
                SP(:,i) = x + (sqrt(n / (1 - W(1)))*P_sqrt(:,i-1));
                SP(:,i+n) = x - (sqrt(n / (1 - W(1)))*P_sqrt(:,i-1));
             end
             
             %GIVE THE OTHER WEIGHTS FOR EACH SIGMA POINT
             W(2:end) = (1 - W(1)) / (2*n);
                       
        case 'CKF'
                   
            % Cubature Rule
             SP = zeros(n,2*n);
             W = zeros(1,2*n);
             
             % Get the other sigma points
             P_sqrt = chol(P,'lower');
             for i = 1:n
                SP(:,i)   = x + sqrt(n)*P_sqrt(:,i);
                SP(:,i+n) = x - sqrt(n)*P_sqrt(:,i);
             end
             %in CUBA no tuning parameter w0, and all weights are the same
             W(:) = 1/(2*n);
        
            
        otherwise
            error('Incorrect type of sigma point')
    end

end


function [x, P] = nonLinKFprediction(x, P, f, Q, type)
%NONLINKFPREDICTION calculates mean and covariance of predicted state
%   density using a non-linear Gaussian model.
%
%Input:
%   x           [n x 1] Prior mean
%   P           [n x n] Prior covariance
%   f           Motion model function handle
%               [fx,Fx]=f(x) 
%               Takes as input x (state), 
%               Returns fx and Fx, motion model and Jacobian evaluated at x
%               All other model parameters, such as sample time T,
%               must be included in the function
%   Q           [n x n] Process noise covariance
%   type        String that specifies the type of non-linear filter
%
%Output:
%   x           [n x 1] predicted state mean
%   P           [n x n] predicted state covariance

    n=length(x);
    switch type
        case 'EKF'       
            % Your EKF code here   
            %Linearize the system by getting the jacobian
            [fx,Fx]=f(x);  %observe x is the prior mean
            x = fx;
            P = Fx*P*transpose(Fx) + Q;
        case 'UKF'
            % Your UKF code here
            % Get the sigma points and the weights
            [SP,W] = sigmaPoints(x, P, 'UKF');
            P=zeros(n,n);
            x=zeros(n,1);
            
            f_nonlin = zeros(n, length(W));
            number_sigma = length(SP);

            for i=1:number_sigma
                f_nonlin(:,i) = f(SP(:,i));
                x=x+f_nonlin(:,i)*W(i);
            end
            for i=1:number_sigma
                P=P+(f_nonlin(:,i)-x)*(f_nonlin(:,i)-x)'*W(i);
            end
             P=P+Q;
             
             
            % Make sure the covariance matrix is semi-definite
            if min(eig(P))<=0
                [v,e] = eig(P, 'vector');
                e(e<0) = 1e-4;
                P = v*diag(e)/v;
            end
                
        case 'CKF'
            
            % Your CKF code here
            % Get the sigma points and the weights
            [SP,W] = sigmaPoints(x, P, 'CKF');
            P=zeros(n,n);
            x=zeros(n,1);
            
            f_nonlin = zeros(n, length(W));
            number_sigma = length(SP);

            for i=1:number_sigma
                f_nonlin(:,i) = f(SP(:,i));
                x=x+f_nonlin(:,i)*W(i);
            end
            for i=1:number_sigma
                P=P+(f_nonlin(:,i)-x)*(f_nonlin(:,i)-x)'*W(i);
            end
             P=P+Q;
             
        otherwise
            error('Incorrect type of non-linear Kalman filter')
    end

end


function [x, P] = nonLinKFupdate(x, P, y, h, R, type)
%NONLINKFUPDATE calculates mean and covariance of predicted state
%   density using a non-linear Gaussian model.
%
%Input:
%   x           [n x 1] Prior mean
%   P           [n x n] Prior covariance
%   y           [m x 1] measurement vector
%   h           Measurement model function handle
%               [hx,Hx]=h(x) 
%               Takes as input x (state), 
%               Returns hx and Hx, measurement model and Jacobian evaluated at x
%               Function must include all model parameters for the particular model, 
%               such as sensor position for some models.
%   R           [m x m] Measurement noise covariance
%   type        String that specifies the type of non-linear filter
%
%Output:
%   x           [n x 1] updated state mean
%   P           [n x n] updated state covariance
%

    n=length(x);
    m=length(y);
    switch type
        
        case 'EKF'
            [hx,Hx]=h(x);
            S=Hx*P*Hx'+R;
            K=P*Hx'*inv(S);

            %outputs
            x=x+K*(y-hx);
            P=P-K*S*K'; % P is prior variance
        case 'UKF'
    
            % Your UKF update here
            [SP,W] = sigmaPoints(x, P, 'UKF');
            %innitialization 
            y_predicted_est=zeros(m,1); %predicted measurment mean
            Pxy=zeros(n,m);%cross covariance
            sigma_transformed = zeros(m, length(W));
            S=R;
            for i=1:length(SP)
                sigma_transformed(:,i)= h(SP(:,i));
                y_predicted_est=y_predicted_est+sigma_transformed(:,i)*W(i);
            end
            for i=1:length(SP)
                S=S+(sigma_transformed(:,i)-y_predicted_est)*(sigma_transformed(:,i)-y_predicted_est)'*W(i);
                Pxy=Pxy+(SP(:,i)-x)*(sigma_transformed(:,i)-y_predicted_est)'*W(i);

            end
            x=x+Pxy*inv(S)*(y-y_predicted_est);
            P=P-Pxy*inv(S)*Pxy';

            
            % Make sure the covariance matrix is semi-definite
            if min(eig(P))<=0
                [v,e] = eig(P, 'vector');
                e(e<0) = 1e-4;
                P = v*diag(e)/v;
            end
            
        case 'CKF'
    
            % Your CKF update here
            [SP,W] = sigmaPoints(x, P, 'CKF');
            %innitialization 
            y_predicted_est=zeros(m,1); %predicted measurment mean
            sigma_transformed = zeros(m, length(W));
            Pxy=zeros(n,m);
            S=R;
            for i=1:length(SP)
                sigma_transformed(:,i)= h(SP(:,i));
                y_predicted_est=y_predicted_est+sigma_transformed(:,i)*W(i);
            end
            for i=1:length(SP)
                S=S+(sigma_transformed(:,i)-y_predicted_est)*(sigma_transformed(:,i)-y_predicted_est)'*W(i);
                Pxy=Pxy+(SP(:,i)-x)*(sigma_transformed(:,i)-y_predicted_est)'*W(i);

            end
            x=x+Pxy*inv(S)*(y-y_predicted_est);
            P=P-Pxy*inv(S)*Pxy';


        otherwise
            error('Incorrect type of non-linear Kalman filter')
    end

end



function [xf, Pf, xp, Pp] = nonLinearKalmanFilter(Y, x_0, P_0, f, Q, h, R, type)
%NONLINEARKALMANFILTER Filters measurement sequence Y using a 
% non-linear Kalman filter. 
%
%Input:
%   Y           [m x N] Measurement sequence for times 1,...,N
%   x_0         [n x 1] Prior mean for time 0
%   P_0         [n x n] Prior covariance
%   f                   Motion model function handle
%                       [fx,Fx]=f(x) 
%                       Takes as input x (state) 
%                       Returns fx and Fx, motion model and Jacobian evaluated at x
%   Q           [n x n] Process noise covariance
%   h                   Measurement model function handle
%                       [hx,Hx]=h(x,T) 
%                       Takes as input x (state), 
%                       Returns hx and Hx, measurement model and Jacobian evaluated at x
%   R           [m x m] Measurement noise covariance
%
%Output:
%   xf          [n x N]     Filtered estimates for times 1,...,N
%   Pf          [n x n x N] Filter error convariance
%   xp          [n x N]     Predicted estimates for times 1,...,N
%   Pp          [n x n x N] Filter error convariance
%

% Your code here. 

% Number of states
n = length(x_0);
% Number of measurements
N = size(Y,2);

% Initialize outputs
xp = zeros(n,N);  %store each time instant in it that is why we have N
Pp = zeros(n,n,N); %store each time instant in it
xf = zeros(n,N); 
Pf = zeros(n,n,N);


% Iterate through the measurements
for i =1:N
    % Prediction
    [x_predicted, P_predicted] = nonLinKFprediction(x_0, P_0, f, Q, type);
    Pp(:,:,i) = P_predicted;
    xp(:,i) = x_predicted;

    % Update STEP and update x_0 and P_0 for next iteration
    [x_0, P_0] = nonLinKFupdate(x_predicted, P_predicted, Y(:,i), h, R, type);
    Pf(:,:,i) = P_0;
    xf(:,i) = x_0;


end
end
